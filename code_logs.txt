export const measurePerformance = () => {
  if (process.env.NODE_ENV === "development") {
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      entries.forEach((entry) => {
        if (entry.duration > 100) {
          const taskInfo = {
            type: entry.entryType,
            name: entry.name || "Unknown Task",
            duration: Math.round(entry.duration),
            startTime: Math.round(entry.startTime),
            component: entry.name.includes("chunk-") ? "Code Chunk Loading" : entry.name.includes("framer-motion") ? "Animation" : entry.name.includes("react") ? "React Processing" : entry.name === "self" ? "Main Thread Work" : "Other Task",
            severity: entry.duration > 150 ? "High" : "Medium"
            // Adjusted severity threshold
          };
          if (!taskInfo.name.includes("chunk-") || taskInfo.duration > 200) {
            console.warn(`Long Task (${taskInfo.severity} Impact):`, taskInfo);
          }
          if (taskInfo.duration > 200) {
            console.trace("Long Task Stack Trace:");
          }
        }
      });
    });
    let frameCount = 0;
    let lastTime = performance.now();
    let lowFPSCount = 0;
    let animationFrameId;
    let isInitialLoad = true;
    setTimeout(() => {
      isInitialLoad = false;
    }, 5e3);
    const checkFPS = () => {
      const now = performance.now();
      frameCount++;
      if (now - lastTime > 1e3) {
        const fps = Math.round(frameCount * 1e3 / (now - lastTime));
        if (fps < 30 && !isInitialLoad) {
          lowFPSCount++;
          console.warn("Performance Alert:", {
            issue: "Low FPS",
            fps,
            consecutive: lowFPSCount,
            timestamp: (/* @__PURE__ */ new Date()).toISOString(),
            memory: getMemoryInfo()
          });
        } else {
          lowFPSCount = 0;
        }
        frameCount = 0;
        lastTime = now;
      }
      animationFrameId = requestAnimationFrame(checkFPS);
    };
    const getMemoryInfo = () => {
      const memory = performance.memory;
      if (!memory) return "Not available";
      return {
        used: Math.round(memory.usedJSHeapSize / 1024 / 1024) + "MB",
        total: Math.round(memory.totalJSHeapSize / 1024 / 1024) + "MB",
        limit: Math.round(memory.jsHeapSizeLimit / 1024 / 1024) + "MB"
      };
    };
    try {
      observer.observe({
        entryTypes: ["longtask", "resource", "measure", "layout-shift", "largest-contentful-paint"]
      });
      animationFrameId = requestAnimationFrame(checkFPS);
      setTimeout(() => {
        console.log("Performance monitoring initialized", {
          timestamp: (/* @__PURE__ */ new Date()).toISOString(),
          environment: process.env.NODE_ENV,
          initialMemory: getMemoryInfo()
        });
      }, 2e3);
      return () => {
        observer.disconnect();
        cancelAnimationFrame(animationFrameId);
      };
    } catch (error) {
      console.error("Performance monitoring setup failed:", error);
    }
  }
  return () => {
  };
};
